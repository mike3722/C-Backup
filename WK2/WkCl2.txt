void readNums()
{
	int num, total = 0, count = 0;

	ifstream in("C:\\Users\\D00167295\\Desktop\\C++WK2\\q3.txt");

	while (!in.eof())
	{
		in >> num;
		total += num;
		count++;
	}

	double avg = (float)total / (float)count;
	cout << "Average = " << avg << endl;

}
